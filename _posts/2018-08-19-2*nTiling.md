## 11726. 2*n 타일링



```
#include <iostream>
using namespace std;

int n;
int dp[1001];

void in() {
    cin >> n;
    dp[0] = 1;
    dp[1] = 1;
}

void tiling() {
    for (int i = 2; i <= n; i++) {
        dp[i] = (dp[i-1] + dp[i-2]) % 10007;
    }
}

void out() {
    cout << dp[n] << endl;
}

int main(int argc, const char * argv[]) {
    in();
    tiling();
    out();
    return 0;
}

```
